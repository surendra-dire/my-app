#/etc/nginx/sites-available/default

server {
    listen 80;
    server_name _;

    root /var/www/react;
    index index.html;

    location / {
        try_files $uri /index.html;
    }

    location /api/ {
        proxy_pass http://localhost:8080;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }
}
